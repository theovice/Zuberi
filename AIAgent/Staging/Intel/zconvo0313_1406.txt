3. Typical tool implementation
# office_io.py
import json
import sys
from pathlib import Path
from openpyxl import load_workbook, Workbook
from docx import Document
from pptx import Presentation

def read_docx(path):
    doc = Document(str(path))
    return {"text": "\n".join(p.text for p in doc.paragraphs)}

def write_docx(path, content):
    doc = Document()
    doc.add_paragraph(content.get("text", ""))
    doc.save(str(path))
    return {"status":"ok"}

def read_xlsx(path):
    wb = load_workbook(str(path), read_only=True)
    data = {}
    for sh in wb.sheetnames:
        sheet = wb[sh]
        data[sh] = [[c.value for c in r] for r in sheet.iter_rows()]
    return data

def write_xlsx(path, data):
    wb = Workbook()
    for sh, rows in data.items():
        ws = wb.active
        ws.title = sh
        for r in rows:
            ws.append(r)
    wb.save(str(path))
    return {"status":"ok"}

def main():
    cmd, *args = sys.argv[1:]
    p = Path(args[0])
    if cmd == "read":
        if p.suffix in {".docx"}:
            print(json.dumps(read_docx(p)))
        elif p.suffix in {".xlsx"}:
            print(json.dumps(read_xlsx(p)))
        else:
            print(json.dumps({"error":"unsupported format"}))
    elif cmd == "write":
        if p.suffix in {".docx"}:
            content = json.loads(args[1])
            print(json.dumps(write_docx(p, content)))
        elif p.suffix in {".xlsx"}:
            data = json.loads(args[1])
            print(json.dumps(write_xlsx(p, data)))
        else:
            print(json.dumps({"error":"unsupported format"}))
    else:
        print(json.dumps({"error":"unknown command"}))

if __name__ == "__main__":
    main()
